41  write(logu,*) ""

44  write(logu,*) "============================================"

63  write(logu,'(1x,a,i0)') "Last iteration = ", it
64  write(logu,'(1x,a,es22.15)') "Current time = ", time

74  if (datadir(l:l).ne.'/') then

85

88  unitin = 10 + rank
89  open (unit=unitin, file=blocksfile, status='old', access='stream', iostat=istat)
90  if (istat.ne.0) then
91  write(logu,'(a,a,a)') "Could not open the file '", trim(blocksfile), "' !"
92  write(logu,'(a,a,a)') "Does the datadir '", trim(datadir_old), "' exist?"
93  close(unitin)
94  call clean_abort (error_warm_file_open)
95  end if
96 
97  ! Read data
98  localblocks(:) = -1
99  nblocks = 0
100  read(unitin) nblocal
101  do nb=1,nblocal
102  read(unitin) localblocks(nb)
103  write(logu,'(1x,a,i0,a)') "Loading block ", localblocks(nb), " ..."
104  read(unitin) u(nb,:,1:ncells_x,1:ncells_y,1:ncells_z)
105  nblocks = nblocks + 1
106  end do
107 
108  if (nblocal.ne.nblocks) then
109  write(logu,'(a,i0,a,i0,a)') "Read ", nblocks, " blocks from file, but expected ", nblocal, " !"
110  write(logu,'(a)') "***Aborting***"
111  call clean_abort (error_warm_read_blocks)
112  else
113  write(logu,'(1x,a,i0,a,i0,a)') "Loaded ", nblocks, " blocks; expected ", nblocal, "."
114  end if
115 
116  close (unitin)
117 
118 end subroutine warmstart
