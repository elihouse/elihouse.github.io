26 !> @brief Sets the initial flow conditions
27 !> @details This wrapper subroutine is called to set the initial flow
28 !! conditions on the mesh. It will first initialize the grid with a
29 !! uniform IC and then call userIC(), where user-defined custom
30 !! initial conditions can be specified.

32

43 
44  write(logu,'(1x,a)') "> Initializing ambient medium ..."
45  write(logu,*) ""
46

49

60 !> @brief Sets a uniform IC on all blocks
61 !> @details The initial medium is uniformly filled with gas having
62 !! the properties defined in the parameters module.
63 subroutine uniformic ()
64 
65  use parameters
66  use globals
67  implicit none
68 
69  integer :: i, j, k, nb, bID
70  real :: primit(neqtot), uvars(neqtot)
71  real :: ism_pres
72 
73  ! ISM is assumed neutral
74  ism_pres = ism_dens/(ism_mu0*amu)*kb*ism_temp
75 
76  write(logu,'(1x,a)') "Ambient medium parameters:"
77  write(logu,'(1x,a,es12.5,a,es12.5,a)') "Dens= ", ism_dens, " g cm^-3 (", ism_dens/d_sc, " code units)"
78  write(logu,'(1x,a,es12.5,a,es12.5,a)') "Pres= ", ism_pres, " dyn cm^-2 (", ism_pres/p_sc, ")"
79  write(logu,'(1x,a,es12.5,a)') "Temp= ", ism_temp, " K (no code units)"
80  write(logu,'(1x,a,es12.5,a,es12.5,a)') "Velx= ", ism_vx, " cm s^-1 (", ism_vx/p_sc, ")"
81  write(logu,'(1x,a,es12.5,a,es12.5,a)') "Vely= ", ism_vy, " cm s^-1 (", ism_vy/p_sc, ")"
82  write(logu,'(1x,a,es12.5,a,es12.5,a)') "Velz= ", ism_vz, " cm s^-1 (", ism_vz/p_sc, ")"
83 #ifdef PASBP
84  write(logu,'(1x,a,es12.5,a)') "Bx= ", ism_bx, " G"
85  write(logu,'(1x,a,es12.5,a)') "By= ", ism_by, " G"
86  write(logu,'(1x,a,es12.5,a)') "Bz= ", ism_bz, " G"
87 #endif
88 
89  ! For all local blocks ...
90  do nb=1,nbmaxproc
91  bid = localblocks(nb)
92  if (bid.ne.-1) then
93 
94  ! Set all physical cells
95  do i=1,ncells_x
96  do j=1,ncells_y
97  do k=1,ncells_z
98 
99  ! Calculate scaled primitives for ISM

103  primit(4) = ism_vz/v_sc

107  primit(7) = ism_by
108  primit(8) = ism_bz
109 #endif
110

114  end if
115 
116  ! Passive scalar for metalicity, for COOL_TABLE_METAL
117  if (cooling_type.eq.cool_table_metal) then
118  primit(metalpas) = ism_metal*primit(1)
119  end if
120 
121  ! Convert primitives and set flow vars for this cell
122  call prim2flow (primit, uvars)
123  u(nb,:,i,j,k) = uvars(:)
124 
125  end do
126  end do
127  end do

132 
133  end if

136 end subroutine uniformic
