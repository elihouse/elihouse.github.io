27 !> @brief High-level wrapper routine for boundary conditions
28 !> @details This is a wrapper routine that calls the two boundary-passing
29 !! routines: normalBoundary, which applies standard boundary conditions,
30 !! and userBoundary, which applies user-defined custom conditions (this
31 !! routine is provided by user.f90, a file which must be modified
32 !! by the user on a per-problem basis).
33 !> @param depth The number of layers of ghost cells that are to be passed
34 !> @param uvars Vector of flow variables to be modified
35 subroutine boundary (depth, uvars)
36 
37  use parameters
38  use userconds
39  implicit none
40 
41  integer, intent(in) :: depth
42  real, intent(inout) :: uvars(nbMaxProc, neqtot, &
43  nxmin:nxmax, nymin:nymax, nzmin:nzmax)
44 
45  ! Apply standard boundary conditions on ghost cells
46  call normalboundary (depth, uvars)
47  call mpi_barrier (mpi_comm_world, ierr) ! This barrier might not be needed
48 
49  ! Apply user-defined boundary conditions (provided by user.f90)
50  call userboundary (uvars)
51  call mpi_barrier (mpi_comm_world, ierr) ! This barrier might not be needed
52 
53 end subroutine boundary
