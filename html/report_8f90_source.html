26 !> @brief Prints out the "main" progress report (after each iteration)
27 subroutine main_report ()
28 
29  use parameters
30  use globals
31  use tictoc
32  implicit none
33 
34  character(30) :: timestr, dtstr
35 
36  call nicetime(time*t_sc, timestr)
37  call nicetime(dt*t_sc, dtstr)
38 
39  ! Report progress
40  write(logu,*) "============================================"
41  write(logu,'(1x,a,i0,a)') "Iteration " , it, " complete!"
42  write(logu,'(1x,a)') stamp()
43  write(logu,'(1x,a,a)') "Elapsed: ", nicetoc(start_mark)
44  write(logu,'(1x,a,a)') "time = ", trim(timestr)
45  write(logu,'(1x,a,a)') "dt = ", trim(dtstr)
46  write(logu,'(1x,i0,a)') nblocal, " local blocks"
47  write(logu,*) "============================================"
48  write(logu,'(1x,a,i0,a,a,a,a,a,i0,a,i0)') "it ", it, "; t= ", &
49  trim(timestr), "; dt=", trim(dtstr), "; blocks=", nblocal, "/", nbactive
50  write(logu,*) "============================================"
51  write(logu,*) ""
52 
53 end subroutine main_report
