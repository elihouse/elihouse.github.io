26 !> @brief High-level wrapper for the numerical integrator
27 subroutine hydrosolver ()
28 
29  use parameters
30  use globals
31  use tictoc
32  implicit none
33 
34  integer :: mark, mark1
35  real :: dt_new
36 
37  call tic(mark)
38  write(logu,*) ""
39  write(logu,'(1x,a)') "============================================"
40  write(logu,'(1x,a)') " Integrating hydrodynamical equations ..."
41  write(logu,'(1x,a)') "============================================"
42  write(logu,*) ""
43 
44  ! Compute global timestep
45  call tic(mark1)
46  write(logu,'(1x,a)') "> Calculating timestep ..."
47  call gettimestep (dt_new)
48  if ((dt.gt.0.0).and.(abs(dt-dt_new)/dt.gt.50)) then
49  write(logu,'(a)') "Abnormal timestep change!"
50  write(logu,'(a,es12.5)') "Previous timestep:", dt
51  write(logu,'(a,es12.5)') "New timestep: ", dt_new
52  write(logu,'(a)') "***ABORTING***"
53  call clean_abort(error_timestep)
54  end if
55  dt = dt_new
56  write(logu,'(1x,a,es12.5,a)') "dt= ", dt, " code units"
57  write(logu,'(1x,a,f12.5,a)') "dt= ", dt*t_sc/yr, " yr"
58  write(logu,'(1x,a,a)') "Timestep calculated in ", nicetoc(mark1)
59 
60  ! Call appropriate numerical solver
61  ! Once this routine returns, the UPs are expected to have correct
62  ! timestepped values in all physical cells (ghost cells are unimportant)
63  call tic(mark1)
64 
65  select case (solver_type)
66 
67  case (solver_lax)
68  call laxfriedrichs ()
69 
70  case (solver_hll1)
71  call godunov (1)
72 
73  case (solver_hll)
74  call godunov (2)
75 
76  case (solver_hllc)
77  call godunov (2)
78 
79  end select
80 
81  write(logu,*) ""
82  write(logu,'(1x,a,a)') "Integrator (total)=", nicetoc(mark1)
83 
84  ! Step hydro variables
85  call tic(mark1)
86  call dostep()
87  write(logu,*) "Stepping=", nicetoc(mark1)
88 
89  ! Done
90  write(logu,*) ""
91  write(logu,'(1x,a,a)') "> Numerical integration completed in ", nicetoc(mark)

97 !> @brief Obtains the simulation-wide numerical timestep
98 !> @details Will compute all the relevant timesteps (hydrodynamical, cooling,
99 !! etc) and return the largest, modified by the CFL parameter
100 !> @param dt_glob The global timestep
101 subroutine gettimestep (dt_glob)
102 
103  use parameters
104  use globals
105  implicit none
106 
107  real, intent(out) :: dt_glob
108 
109  integer :: nb, bID, i, j, k, ilev
110  real :: cs, dt_hydro, dt_cool, dt_loc
111 
112  dt_loc = 1.0e30
113  dt_glob = 1.0e30
114  dt_hydro = 1.0e30
115  dt_cool = 1.0e30
116 
117  do nb=1,nbmaxproc
118  bid = localblocks(nb)
119  if (bid.ne.-1) then
120 
121  call meshlevel(bid, ilev)
122 
123  do i=1,ncells_x
124  do j=1,ncells_y
125  do k=1,ncells_z
126 
127  ! Hydrodynamical timestep
128  call sound (prim(nb,:,i,j,k),cs)
129  dt_hydro = min( dt_hydro, &
130  dx(ilev)/(abs(prim(nb,2,i,j,k))+cs), &
131  dy(ilev)/(abs(prim(nb,3,i,j,k))+cs), &
132  dz(ilev)/(abs(prim(nb,4,i,j,k))+cs) )
133 
134  ! Cooling timestep - not needed at the moment
135 
136  end do
137  end do
138  end do
139 
140  end if
141  end do
142 
143  ! Limit the timestep (optional)
144  if (cooling_type.eq.cool_none) then
145  dt_loc = dt_hydro*cfl
146  else
147  dt_loc = min( dt_hydro, dt_cool ) * cfl
148  end if
149 
150 #ifdef MPIP
151  ! Obtain the global minimum timestep through MPI
152  call mpi_allreduce(dt_loc, dt_glob, 1, mpi_real_kind, mpi_min, mpi_comm_world, ierr)
153 #else
154  dt_glob = dt_loc
155 #endif
156 
157  return
158 
159 end subroutine gettimestep
