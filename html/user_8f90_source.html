26 ! EXAMPLE: winds
27 ! This example shows how to impose wind sources, both plane and spherical.
28 
29 ! In this example we simulate the bow shock produced when a spherical wind
30 ! source moves through the ISM. This is achieved by imposing a spherical wind
31 ! source at the center of a 1x1x1 pc box as well as a planar wind incoming
32 ! from one of the edges of the box.
33 
34 ! More specifically, we do three things:
35 ! 1) The initial ISM is set in motion initially.
36 ! 2) As part of the initial conditions, we impose both the spherical wind
37 ! source and the planar wind source (coming in from the top of the
38 ! computational box moving towards -Z).
39 ! 3) At each boundary condition exchange, these wind sources are refreshed.
40 
41 ! The wind sources are imposed by calling subroutines from the winds module.
42 ! These routines must be given a wind parameters object (of the correspodning
43 ! type) which holds all the parameters of the wind.
44 
45 module userconds
46 ! ============================================
47 !< @brief User-specified initial and boundary conditions
48 !< @details This is the file where the user sets the initial conditions of
49 !! the simulation, as well as any special boundary conditions or custom
50 !! modifications to the simulation.
51 !!
52 !! How to use this module:
53 !! 1) Add "use" statements to include any modules you require in the
54 !! section marked [1].
55 !! 2) You can define aditional module-wide global variables that you
56 !! may require in the section marked [2].
57 !! 3) Fill in the subroutine userInitialCondition(), marked [3], which
58 !! is called at the beginning of the simulation.
59 !! 4) Optionally, fill in the subroutine userBoundary(), marked [4],
60 !! which is called at the end of each boundary exchange operation.
61 !! All subroutines in this module automatically have access to the global
62 !! parameters and variables.
63 ! ============================================
64

66  use globals

79  ! We declare the two wind parameters objects

87  subroutine userinitialcondition (uvars)
88  ! ============================================
89  ! [3] USER-DEFINED INITIAL CONDITIONS
90  !
91  !< @brief User-defined Initial Conditions
92  !< @details This subroutine is called at the beginning of the simulation,
93  !! after the base grid is built and a basic uniform initial condition
94  !! is imposed. It is to be modified by the user to define the problem-
95  !! specific Initial Condition.
96  !!
97  !! IMPORTANT: This subroutine receives the FLOW variables array to be
98  !! modified as argument 'uvars'. The subroutine must modify this array,
99  !! *NOT* the global arrays U, UP or PRIMS.
100  !!
101  !! The array has the following structure:
102  !! uvars ( block ID, equation number, cell_i, cell_j, cell_k )
103  !! where the equation numbering is:
104  !! 1: rho
105  !! 2: rho*u
106  !! 3: rho*v
107  !! 4: rho*w
108  !! 5: E (kinetic+thermal)
109  !! If the passive magnetic field is enabled:
110  !! 6: B_x
111  !! 7: B_y
112  !! 8: B_z
113  !! If passive scalars are enabled, they begin after all other flow
114  !! variables (i.e., 9+ if passive B-field enabled, 6+ if not).
115  !!
116  !! Note that the cell indices include ghost cells. For instance, if we
117  !! had a 1-deep ghost cell layer then cells from 1 to ncells_x would be
118  !! physical cells, while cells 0 and ncells_x+1 would be ghost cells.
119  ! ============================================
120 
121  implicit none
122  real, intent(inout) :: uvars (nbMaxProc, neqtot, &
123  nxmin:nxmax, nymin:nymax, nzmin:nzmax)
124 
125  ! ============================================
126 
127  ! First we fill in the wind parameters
128  spherical_wind%xc = 0.5 * pc
129  spherical_wind%yc = 0.5 * pc
130  spherical_wind%zc = 0.5 * pc
131  spherical_wind%radius = 0.1 * pc
132  spherical_wind%mdot = 3e-7 * msun/yr
133  spherical_wind%vinf = 5 * kps
134  spherical_wind%temp = 100
135  spherical_wind%mu = 1.3
136 
137  ! For the plane wind we just copy the ISM parameters
138  plane_wind%plane = plane_top
139  plane_wind%rho = ism_dens
140  plane_wind%vel = abs(ism_vz)
141  plane_wind%temp = ism_temp
142  plane_wind%mu = ism_mu0
143 
144  ! Then we impose the wind sources
145  call imposesphericalwind (spherical_wind, uvars)
146  call imposeplanewind (plane_wind, uvars)
147 
148  ! ============================================
149 
150  end subroutine userinitialcondition
151

154  subroutine userboundary (uvars)
155  ! ============================================
156  ! [4] USER-DEFINED BOUNDARY CONDITIONS
157  !
158  !< @brief User-defined Boundary Conditions
159  !< @details This subroutine is called once per timestep *after* standard
160  !! boundary have been applied to all blocks. It allows the user to
161  !! to impose an arbitrary boundary condition on the simulation.
162  !!
163  !! IMPORTANT: This subroutine receives the FLOW variables array to be
164  !! modified as argument 'uvars'. The subroutine must modify this array,
165  !! *NOT* the global arrays U and UP.
166  !!
167  !! The structure of this array is described in the userInitialConditions()
168  !! subroutine documentation above.
169  ! ============================================
170 
171  implicit none
172  real, intent(inout) :: uvars (nbMaxProc, neqtot, &
173  nxmin:nxmax, nymin:nymax, nzmin:nzmax)
174  ! ============================================
175 
176  ! After each boundary exchange, we refresh the wind sources
177  call imposesphericalwind (spherical_wind, uvars)
178  call imposeplanewind (plane_wind, uvars)
179 
180  ! ============================================
181 
182  end subroutine userboundary
183 
184 end module userconds
