26 !> @brief Initializes root blocks and refines the mesh to prepare it for ICs
27 !> @details This is a main-level routine. Calculates the mesh geometry
28 !! (how many root blocks) and number of levels to meet the required resolution,
29 !! and refines the IC zone.
30 subroutine basegrid ()
31 
32  use parameters
33  use globals
34  use tictoc
35  implicit none
36 
37  ! Local variables
38  integer :: maxlevx, maxlevy, maxlevz, smalldim
39  integer :: nb, bID, ilev, x, y, z, mark
40  real :: smallsize
41 
42  call tic(mark)
43 
44  write(logu,*) ""
45  write(logu,*) "============================================"
46  write(logu,'(1x,a)') " Initializing the base grid ..."
47  write(logu,*) "============================================"
48  write(logu,*) ""
49 
50  ! Every rank calculates the base grid geometry, based on the selected method
51  if (mesh_method.eq.mesh_auto) then
52 
53  maxcells_x = p_maxcells_x
54  maxcells_y = p_maxcells_y
55  maxcells_z = p_maxcells_z
56 
57  ! Number of root blocks
58  smalldim = min(maxcells_x,maxcells_y,maxcells_z)
59  nbrootx = maxcells_x / smalldim
60  nbrooty = maxcells_y / smalldim
61  nbrootz = maxcells_z / smalldim
62 
63  ! Calculate number of levels
64  maxlevx = int(log(float(maxcells_x)/nbrootx/ncells_x)/log(2.)+1)
65  maxlevy = int(log(float(maxcells_y)/nbrooty/ncells_y)/log(2.)+1)
66  maxlevz = int(log(float(maxcells_z)/nbrootz/ncells_z)/log(2.)+1)
67  maxlev = max(maxlevx,maxlevy,maxlevz)
68 
69 
70  else if (mesh_method.eq.mesh_manual) then
71 
72  maxlev = p_maxlev
73  nbrootx = p_nbrootx
74  nbrooty = p_nbrooty
75  nbrootz = p_nbrootz
76 
77  ! Number of max-resolution cells
78  maxcells_x = nbrootx * ncells_x * 2**(maxlev-1)
79  maxcells_y = nbrooty * ncells_y * 2**(maxlev-1)
80  maxcells_z = nbrootz * ncells_z * 2**(maxlev-1)
81  smalldim = min(maxcells_x, maxcells_y, maxcells_z)
82 
83  end if
84 
85  ! The maximum allowable number of levels is 10 at the moment, due to the
86  ! fact that bIDs are handled as 4-byte (signed) integers. With 10 levels,
87  ! one can have up to 14 root blocks.
88  if (maxlev.gt.10) then
89  write(logu,*) ""
90  write(logu,'(1x,a)') "Desired finest-level resolution would require more than 10 levels!"
91  write(logu,'(1x,a)') "***ABORTING***"
92  call clean_abort (error_too_many_levs)
93  end if
94 
95  ! Check if physical sizes conform to desired max-resolution cell numbers
96  smallsize = min(xphystot, yphystot, zphystot)
97  if ((maxcells_x/smalldim.ne.xphystot/smallsize).or.&
98  (maxcells_y/smalldim.ne.yphystot/smallsize).or.&
99  (maxcells_z/smalldim.ne.zphystot/smallsize)) then
100  write(logu,*)
101  write(logu,'(a)') "ERROR: The physical size of the simulation box has"//&
102  " a different aspect ratio than the requested number of cells!"
103  write(logu,'(a,f4.1,a,f4.1,a,f4.1)') "Physical size aspect ratio: ",&
104  xphystot/smallsize, " : ", yphystot/smallsize, " : ", zphystot/smallsize
105  write(logu,'(a,i3,a,i3,a,i3)') "Cell number aspect ratio: ",&
106  maxcells_x/smalldim, " : ", maxcells_y/smalldim, " : ", maxcells_z/smalldim
107  write(logu,'(a)') "> Modify parameters.f90"
108  call clean_abort (error_basegrid_bad_aspect)
109  else
110  write(logu,'(1x,a,i2,a,i2,a,i2)') "Grid geometry (root blocks): ", &
111  nbrootx, " x ", nbrooty, " x ", nbrootz
112  write(logu,'(1x,a,i3)') "Number of root blocks: ", nbrootx*nbrooty*nbrootz
113  write(logu,'(1x,a,i2)') "Number of refinement levels: ", maxlev
114  end if
115 
116  ! Allocate and initialize grid spacing at each level (code units)
117  allocate( dx(maxlev) )
118  allocate( dy(maxlev) )
119  allocate( dz(maxlev) )
120  do ilev=1,maxlev
121  dx(ilev) = xphystot / (ncells_x * nbrootx * 2.0**(ilev-1)) / l_sc
122  dy(ilev) = yphystot / (ncells_y * nbrooty * 2.0**(ilev-1)) / l_sc
123  dz(ilev) = zphystot / (ncells_z * nbrootz * 2.0**(ilev-1)) / l_sc
124  end do
125 
126  ! Calculate total number of blocks per level
127  allocate( nblockslev(maxlev) )
128  do ilev=1,maxlev
129  nblockslev(ilev) = nbrootx*nbrooty*nbrootz*8**(ilev-1)
130  end do
131 
132  ! Calculate number of blocks along each dimension at each level
133  allocate( nbx(maxlev) )
134  allocate( nby(maxlev) )
135  allocate( nbz(maxlev) )
136  do ilev=1,maxlev
137  nbx(ilev) = nbrootx*2**(ilev-1)
138  nby(ilev) = nbrooty*2**(ilev-1)
139  nbz(ilev) = nbrootz*2**(ilev-1)
140  end do
141 
142  ! Calculate the bIDs of the first and last blocks at each level
143  allocate( minid(maxlev) )
144  allocate( maxid(maxlev) )
145  minid(1) = 1
146  maxid(1) = nblockslev(1)
147  do ilev=2,maxlev
148  minid(ilev) = maxid(ilev-1) + 1
149  maxid(ilev) = maxid(ilev-1) + nblockslev(ilev)
150  end do
151 
152  nblocal = 0
153  nbactive = nbrootx*nbrooty*nbrootz
154 
155  ! ==========================
156  ! The following is only performed in cold starts
157 
158  if (.not.dowarm) then
159 
160  ! Activate root blocks and initialize block registry
161  write(logu,'(1x,a)') "> Creating root blocks ..."
162 
163  ! Calculate bID of root blocks and register them in master's block list
164  if (rank.eq.master) then
165  nb = 1
166  do z = 1,nbrootz
167  do y = 1,nbrooty
168  do x = 1,nbrootx
169 
170  ! bID is sensitive to numbering order
171  bid = 1+(x-1)+(y-1)*nbrootx+(z-1)*nbrootx*nbrooty
172 
173  ! Register block
174  localblocks(nb) = bid
175 
176  nb = nb + 1
177  nblocal = nblocal + 1
178 
179  end do
180  end do
181  end do
182  end if
183 
184  ! Synchronize block lists
185  call syncblocklists ()
186 
187  else
188 
189  write(logu,*) ""
190  write(logu,'(1x,a)') "> Skipping root block creation (warm start)"
191 
192  end if
193 
194  ! ==========================
195 
196  write(logu,*) ""
197  write(logu,'(1x,a,a)') "> Created base grid in ", nicetoc(mark)
198  write(logu,*) ""
199 
200 end subroutine basegrid
