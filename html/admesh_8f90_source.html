26 !> @brief High-level routine that triggers mesh refinements

120 !> @brief Marks a block for refinement or coarsening if it meets physical criteria
121 !> @details Note: this requires updated primitives
122 !> @param locInd Local index of the block to be checked
123 !> @param bID bID of the block
124 !> @param flag Refinement flag
125 subroutine markbyphysical (locIndx, bID, flag)
126 
127  use parameters
128  use globals
129  implicit none
130 
131  integer, intent(in) :: locIndx
132  integer, intent(in) :: bID
133  integer, intent(out) :: flag
134  integer :: i, j, k, ilev
135  real :: grad, maxgrad, gradx, grady, gradz
136  logical :: verbose
137 
138  ! DEBUG
139  verbose = .false.

141  call meshlevel (bid, ilev)

143  maxgrad = 0.0
144 
145  do i=1,ncells_x
146  do j=1,ncells_y
147  do k=1,ncells_z
148 
149  ! Pressure gradient check
150 
151  gradx = abs(prim(locindx,5,i+1,j,k)-prim(locindx,5,i-1,j,k)) / &
152  prim(locindx,5,i,j,k) / (2*dx(ilev))
153  grady = abs(prim(locindx,5,i,j+1,k)-prim(locindx,5,i,j-1,k)) / &
154  prim(locindx,5,i,j,k) / (2*dx(ilev))
155  gradz = abs(prim(locindx,5,i,j,k+1)-prim(locindx,5,i,j,k-1)) / &
156  prim(locindx,5,i,j,k) / (2*dx(ilev))

158  maxgrad = max( maxgrad, grad )
159

161  ! unless it is already at the maximum level; exit subroutine regardless
162  if (grad.ge.refinethres) then
163  if (ilev.eq.maxlev) then
164  if (verbose) then
165  write(logu,'(a,i8,a)') "Block ", bid, " can't be refined any further - you might wanna increase the number of levels"
166  end if
167  flag = flag_none
168  return
169  else if (ilev.lt.maxlev) then
170  flag = flag_refine
171  return
172  end if
173  end if
174 
175  ! Density gradient check
176 
177  gradx = abs(prim(locindx,1,i+1,j,k)-prim(locindx,1,i-1,j,k)) / &
178  prim(locindx,1,i,j,k) / (2*dx(ilev))
179  grady = abs(prim(locindx,1,i,j+1,k)-prim(locindx,1,i,j-1,k)) / &
180  prim(locindx,1,i,j,k) / (2*dx(ilev))
181  gradz = abs(prim(locindx,1,i,j,k+1)-prim(locindx,1,i,j,k-1)) / &
182  prim(locindx,1,i,j,k) / (2*dx(ilev))
183  grad = max( gradx, grady, gradz )
184  maxgrad = max( maxgrad, grad )
185 
186  ! If the gradient is larger than the refinement threshold, mark the block
187  ! unless it is already at the maximum level; exit subroutine regardless
188  if (grad.ge.refinethres) then
189  if (ilev.eq.maxlev) then
190  if (verbose) then
191  write(logu,'(a,i8,a)') "Block ", bid, " can't be refined any further - you might wanna increase the number of levels"
192  end if
193  flag = flag_none
194  return
195  else if (ilev.lt.maxlev) then
196  flag = flag_refine
197  return
198  end if
199  end if
200 
201  end do
202  end do
203  end do
204 
205  ! If the maximum gradient is smaller than the coarsening threshold, mark
206  ! it for coarsening, unless it is a root block
207  if ((maxgrad.le.coarsethres).and.(ilev.gt.1)) then
208  flag = flag_coarse
209  else
210  flag = flag_none
211  end if
212 
213  return
214
