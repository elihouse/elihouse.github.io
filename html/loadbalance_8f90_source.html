59 !> @brief Performs load balancing among all processors
60 !> @details It first builds a loadOrder of the blocks following a Hilbert walk.
61 !! Then, it builds a loadScheme, which contains a list of all the active
62 !! blocks and the current and new owner for each block. Finally, it applies
63 !! the computed loadScheme.
64 subroutine dobalance ()
65 
66  use parameters
67  use globals
68  use tictoc
69  implicit none
70 
71  integer :: loadOrder(nbMaxGlobal)
72  integer :: numBlocks(0:nProcs-1)
73  integer :: loadScheme(nbActive, 3), tots(nProcs)
74  integer :: nb, nbloc, p, c, bID, nData
75  integer :: old_owner, new_owner, blocks, badcells
76  integer :: mpistatus(MPI_STATUS_SIZE)
77  integer :: mark
78  integer :: sent, received
79  logical :: verbose
80 
81  verbose = .false.
82 
83  ! The goal is to build and apply the new load balancing scheme, which is
84  ! containted in the loadScheme array. Its structure is:
85  ! loadScheme(:,1) = block's bID
86  ! loadScheme(:,2) = block's current owner
87  ! loadScheme(:,3) = block's new owner
88  ! After it's been built, it is used walked one entry at a time to transfer
89  ! blocks between processes
90 
91  call tic(mark)
92 
93  ! First, build an ordered list of blocks according to its position in
94  ! in a Hilbert walk at the finest level
95  loadorder(:) = -1
96  call hilbertorder (loadorder)
97 
98  ! Determine how many blocks each process gets. Process 0 is assigned last.
99  numblocks(:) = floor(real(nbactive)/nprocs)
100  do nb=1,mod(nbactive,nprocs)
101  p = mod(nb,nprocs)
102  numblocks(p) = numblocks(p) + 1
103  end do
104 
105  ! Initialize the loadScheme list using loadOrder and the current owner
106  c = 1
107  do nb=1,nbmaxglobal
108  if (loadorder(nb).ne.-1) then
109  bid = loadorder(nb)
110  call getowner (bid, old_owner)
111  loadscheme(c,1) = bid
112  loadscheme(c,2) = old_owner
113  loadscheme(c,3) = -1
114  c = c + 1
115  end if
116  end do
117 
118  ! Next, build the new loadScheme
119  nb = 1
120  do p=0,nprocs-1
121  blocks = numblocks(p)
122  do c=1,blocks
123  loadscheme(nb,3) = p
124  nb = nb + 1
125  end do
126  end do
127 
128  ! DEBUG
129  if (verbose) then
130  ! Report the loadScheme
131  write(logu,'(1x,a)') "The new load scheme is:"
132  write(logu,'(6x,a)') "bID old owner -> new owner"
133  tots(:) = 0
134  do nb=1,nbactive
135  bid = loadscheme(nb,1)
136  old_owner = loadscheme(nb,2)
137  new_owner = loadscheme(nb,3)
138  write(logu,'(1x,i8,i8,6x,a,i8)') bid, old_owner, "->", new_owner
139  tots(new_owner+1) = tots(new_owner+1) + 1
140  end do
141  write(logu,'(1x,a)') "Summary:"
142  do p=0,nprocs-1
143  write(logu,'(2x,a,i8,a,i4,a)') "Process ", p, " gets ", tots(p+1), " blocks"
144  end do
145  end if
146 
147  ! Everybody stop here
148  call mpi_barrier (mpi_comm_world, ierr)
149 
150  ! DEBUG
151  if (verbose) then
152  write(logu,*) "There are", nbactive, "blocks globally"
153  write(logu,*) "I have", nblocal, "local blocks"
154  end if
155  ! DEBUG
156 
157  ! Finally, distribute the blocks via MPI messages
158  !
159  ! For every block, compare current and new owner.
160  ! Then, one of four things will happen:
161  ! 1) if the old owner is also the new owner, ignore this code
162  ! (this block doesn't change process)
163  ! 2) if they are different, but neither is my rank, ignore this code
164  ! (I'm not involved in this exchange)
165  ! 3) if they are different, and I'm the old owner, MPI_SEND the block to
166  ! the new owner
167  ! 4) if they are different, and I'm the new owner, MPI_RECV the block from
168  ! the old owner
169  call tic(mark)
170  sent = 0
171  received = 0
172  do nb=1,nbactive
173 
174  bid = loadscheme(nb,1)
175  old_owner = loadscheme(nb,2)
176  new_owner = loadscheme(nb,3)
177 
178  if (old_owner.ne.new_owner) then
179  if (rank.eq.old_owner) then
180 
181  ! Transmit this block (flow variables on physical cells)
182 
183  call find (bid, localblocks, nbmaxproc, nbloc)
184  if (nbloc.eq.-1) then
185  write(logu,*) "Couldn't find bID", bid, "in local list!!"
186  write(logu,*) "***ABORTING!***"
187  call clean_abort (error_loadbal_no_local_bid)
188  end if
189 
190  if (verbose) then
191  write(logu,'(1x,a,i8,a,i4,a,i4)') "Transmitting block ", bid, &
192  " with local index ", nbloc, " to process ", new_owner
193  end if
194  ndata = neqtot*ncells_x*ncells_y*ncells_z
195  call mpi_send( u(nbloc, 1:neqtot, 1:ncells_x, 1:ncells_y, 1:ncells_z), &
196  ndata, mpi_real_kind, new_owner, bid, mpi_comm_world, ierr)
197  localblocks(nbloc) = -1
198 
199  sent = sent + 1
200 
201  else if (rank.eq.new_owner) then
202 
203  ! Receive this block (flow variables on physical cells)
204 
205  call find (-1, localblocks, nbmaxproc, nbloc)
206  if (nbloc.eq.-1) then
207  write(logu,*) "Couldn't find a free local slot!"
208  write(logu,*) "***ABORTING!***"
209  call clean_abort (error_loadbal_insufficient_ram)
210  end if
211 
212  if (verbose) then
213  write(logu,'(1x,a,i8,a,i4,a,i4)') "Receiving block ", bid, &
214  " into free local index ", nbloc, " from process ", old_owner
215  end if
216  ndata = neqtot*ncells_x*ncells_y*ncells_z
217  call mpi_recv( u(nbloc, 1:neqtot, 1:ncells_x, 1:ncells_y, 1:ncells_z), &
218  ndata, mpi_real_kind, old_owner, bid, mpi_comm_world, mpistatus, ierr)
219  localblocks(nbloc) = bid
220 
221  ! Update primitives of received block
222  call calcprimsblock (u, prim, nbloc, cells_phys, badcells)
223 
224  received = received + 1
225 
226  end if
227  end if
228 
229  end do
230 
231  ! Re-calculate number of local blocks
232  nblocal = 0
233  do nb=1,nbmaxproc
234  if (localblocks(nb).ne.-1) then
235  nblocal = nblocal + 1
236  end if
237  end do
238 
239  ! Report load balance results
240  write(logu,'(1x,a,i0,a,i0,a)') "Sent ", sent, " blocks; received ", received, " blocks"
241  write(logu,'(1x,a,i0,a)') "Now have ", nblocal, " local blocks"
242 
243  ! Synchronize global block registry
244  call syncblocklists ()
245 
246  ! DEBUG
247  if (verbose) then
248  write(logu,*) "Local blocks after syncBlockLists:"
249  do nb=1,nbmaxproc
250  if (localblocks(nb).ne.-1) then
251  write(logu,*) nb, localblocks(nb)
252  end if
253  end do
254  end if
255 
256 end subroutine dobalance

260 !> @brief Returns a load-balancing ordered list of blocks
261 !> @details The list is obtained by calculating the index of each block
262 !! along a Hilbert curve, and then sorting the block list using the indices
263 !> @param loadOrder The load-balanced ordered list of bIDs
264 subroutine hilbertorder (loadOrder)
265 
266  use parameters
267  use globals
268  implicit none
269 
270  integer, intent(out) :: loadOrder(nbMaxGlobal)
271 
272  integer :: nb, next, bID, istat
273  integer :: hk
274  integer, allocatable :: blocks(:)
275  integer, allocatable :: keys(:)
276 
277  ! Re-calculate nbActive, declare blocks(:) and keys(:), and fill them
278  ! with blocks and Hilbert Keys
279 
280  nbactive = 0
281  do nb=1,nbmaxglobal
282  if (globalblocks(nb).ne.-1) then
283  nbactive = nbactive + 1
284  end if
285  end do
286 
287  allocate( blocks(nbactive), stat=istat)
288  allocate( keys(nbactive), stat=istat)
289  blocks(:) = -1
290  keys(:) = -1
291 
292  ! Build lists of blocks with corresponding Hilbert indices as keys
293  next = 1
294  do nb=1,nbmaxglobal
295  if (globalblocks(nb).ne.-1) then
296  ! Obtain this block's Hilbert key at the finest mesh level
297  bid = globalblocks(nb)
298  call gethilbertkey(bid, hk)
299  blocks(next) = bid
300  keys(next) = hk ! Implicit conversion to integer here
301  next = next + 1
302  end if
303  end do
304 
305  ! Sort 'blocks' list using 'keys' through QuickSort
306  call quicksort (nbactive, blocks, keys, 1, nbactive)
307 
308  ! Fill loadOrder list with sorted blocks list
309  loadorder(:) = -1
310  do nb=1,nbactive
311  loadorder(nb) = blocks(nb)
312  end do
313 
314  ! DEBUG
315 ! if (rank.eq.master) then
316 ! write(logu,*) "Hilbert indexing:"
317 ! do nb=1,nbActive
318 ! bID = loadOrder(nb)
319 ! call bcoords(bID, x, y, z)
320 ! write(logu,*) bID, x, y, z, keys(nb)
321 ! end do
322 ! end if
323  ! DEBUG
324 
325  ! Don't forget to deallocate these arrays :)
326  deallocate (blocks)
327  deallocate (keys)
328 
329 end subroutine hilbertorder

363 !> @param first The index of the first element to be sorted

365 recursive subroutine quicksort (l, list, keys, first, last)
366 
367  implicit none
368  integer, intent(in) :: l
369  integer, intent(inout) :: list(l)
370  integer, intent(inout) :: keys(l)
371  integer, intent(in) :: first
372  integer, intent(in) :: last
373 
374  integer :: pivotindex, newpivotindex
375 
376  ! Only do something if the lists have at least 2 elements
377  if (first.lt.last) then
378 
379  ! Choose a pivot - middle item
380  pivotindex = first + (last-first+1)/2 ! Implicit conversion to integer
381

384 
385  ! Recursively sort each sublist

395 !> @brief In-place auxiliary partition routine for QuickSort
396 !> @param l The size of the lists
397 !> @param list The list of items to be sorted
398 !> @param keys The list of keys used for the sort
399 subroutine partition (l, list, keys, first, last, pivotIndex, storeIndex)
400 
401  implicit none
402  integer, intent(in) :: l
403  integer, intent(inout) :: list(l)
404  integer, intent(inout) :: keys(l)
405  integer, intent(in) :: first
406  integer, intent(in) :: last
407  integer, intent(in) :: pivotIndex
408  integer, intent(out) :: storeIndex
409 
410  integer :: pivotKey
411  integer :: temp, i
412 
413  ! Get the pivot's key
414  pivotkey = keys(pivotindex)
415 
416  ! Swap pivot value/key with item in last position
417  temp = list(last)
418  list(last) = list(pivotindex)
419  list(pivotindex) = temp
420  temp = keys(last)
421  keys(last) = keys(pivotindex)
422  keys(pivotindex) = temp
423 
424  ! Swap items to begginning if key smaller than pivot
425  storeindex = first
426  do i=first,last-1
427  if (keys(i).le.pivotkey) then
428  temp = list(storeindex)
429  list(storeindex) = list(i)
430  list(i) = temp
431  temp = keys(storeindex)
432  keys(storeindex) = keys(i)
433  keys(i) = temp
434  storeindex = storeindex + 1
435  end if
436  end do
437 
438  ! Move pivot to its final position
439  temp = list(last)
440  list(last) = list(storeindex)
441  list(storeindex) = temp
442  temp = keys(last)
443  keys(last) = keys(storeindex)
444  keys(storeindex) = temp
445 
446 end subroutine partition

450 !> @brief Calculates the Hilbert Key of a bID at the finest mesh level
451 !> @details If the block's level is not the finest,
452 !> @param bID The global ID of the block
453 !> @param hk The block's Hilbert key
454 subroutine gethilbertkey (bID, hk)
455 
456  use parameters
457  use globals
458  implicit none
459 
460  integer, intent(in) :: bID
461  integer, intent(out) :: hk
462 
463  integer :: maxnb, order, x, y, z, ilev, xp, yp, zp
464 
465  ! Determine from the base grid geometry the "bounding cube" for the
466  ! Hilbert curve, and hence the order of the curve
467  maxnb = max(nbx(maxlev), nby(maxlev), nbz(maxlev))
468  order = ceiling(log(1.0*maxnb)/log(2.0))
469 
470  ! Obtain the block's local coords (own mesh level) and convert
471  ! them to coords at the finest mesh level
472  call bcoords(bid, x, y, z)
473  call meshlevel(bid, ilev)
474  xp = (x-1)*2**(maxlev-ilev) + 1
475  yp = (y-1)*2**(maxlev-ilev) + 1
476  zp = (z-1)*2**(maxlev-ilev) + 1
477 
478  ! Get the Hilbert Key from the subroutine in hilbert.f90
479  hk = -1
480  call hkey3(xp-1, yp-1, zp-1, order, hk)
481 
482 end subroutine gethilbertkey
