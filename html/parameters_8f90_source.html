75  ! Mesh Geometry

77  !> Method used to specify mesh geometry
78  ! There are currently two ways to specify the mesh geometry:
79  ! 1) MESH_AUTO: the user specifies max-level resolution.
80  ! This will automatically calculate the number of root blocks
81  ! and number of refinement levels
82  ! 2) MESH_MANUAL: the user specifies the number and geometry of the
83  ! root blocks and the number of refinement levels.
84  ! The max-level resolution will then be given by:
85  ! dx(max) = phys_size / (ncells_x*2^(maxlev-1))
86  ! In both cases, the requested grid geometry must match the aspect
87  ! ratio of the physical box dimensions.
88  integer, parameter :: mesh_method = mesh_auto
89 
90  ! Only one of the following group of parameters will be used:
91 
92  ! 1) MESH_AUTO
93  ! Specify the maximum number of cells desired at the highest refinement
94  ! level
95  ! > MUST BE POWERS OF TWO! <
96  integer, parameter :: p_maxcells_x = 128
97  integer, parameter :: p_maxcells_y = 128
98  integer, parameter :: p_maxcells_z = 128
99 
100  ! -- OR --
101 
102  ! 2) MESH_MANUAL
103  ! Specify number of root blocks along each dimension and number of
104  ! refinement levels
105  integer, parameter :: p_nbrootx = 1
106  integer, parameter :: p_nbrooty = 1
107  integer, parameter :: p_nbrootz = 1
108  integer, parameter :: p_maxlev = 5

110  ! Other Mesh Parameters
111 
112  !> Number of cells per block (per dimension)
113  ! This specifies the number of cells per block along each dimension
114  ! (i.e., along each side of the cube that is a block)
115  ! This is a tradeoff between 1) having a few large blocks, which reduces
116  ! the amount of inter-process communication but also makes the mesh
117  ! refinement inefficient (forcing the code to refine large volumes),
118  ! and 2) having a lot of small blocks, which makes the AMR very efficient
119  ! but increases the amount of communication.
120  ! MUST be a power of two. A good value is 16.
121  integer, parameter :: ncells_block = 16
122 
123  ! Refinement / Coarsening thresholds
124  ! These are the maximum relative gradients required to mark a block for
125  ! either refinement or coarsening. Note: refineThres must be larger than
126  ! twice coarseThres. Good values are 0.3 and 0.03.
127  real, parameter :: refinethres = 0.3 !< Refinement threshold
128  real, parameter :: coarsethres = 0.03 !< Coarsening threshold
