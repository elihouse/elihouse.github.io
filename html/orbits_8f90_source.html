61  ! ============================================! at the given time
62  !> @brief Computes the positions and velocities of the bodies
63  !!
64  !> @details Solves the reduced Kepler problema and returns the
65  !! positions and velocities of the two bodies. Note that the orbital
66  !! phase is defined as zero at periapsis and 0.5 at apoapsis. The
67  !! phase is easily calculated from the time as:
68  !! phase = time/orbital_period + initial_phase
69  !> @params phase The orbital phase at which the solution is to be computed
70  !> @params x1,y1,x2,y2 Positions of the two bodies
71  !> @params vx1,vy1,vx2,vy2 Velocities of the two bodies
72  subroutine computebinary (phase, x1, y1, x2, y2, vx1, vy1, vx2, vy2)
73 
74  implicit none
75 
76  real, intent(in) :: phase
77  real, intent(out) :: x1, y1, x2, y2, vx1, vy1, vx2, vy2
78 
79  real, parameter :: PI = acos(-1.0)
80 
81  real :: mu, a, L, M, E
82  real :: rad, theta, thetadot, raddot
83  real :: radx, rady, vx, vy
84 
85  ! 1) Solve the reduced problem
86 
87  ! Compute reduced mass of the system, semi-major axis of reduced orbit
88  ! and total angular momentum of the reduced mass
89  mu = (m1*m2)/(m1+m2)
90  a = (p*p*grav*(m1+m2)/(4*pi*pi))**(1.0/3.0)
91  l = mu*(1-ecc)*a*sqrt((1+ecc)/(1-ecc)*grav*(m1+m2)/a)
92 
93  ! Mean anomaly M
94  m = 2*pi*phase
95 
96  ! Obtain eccentric anomaly E by numerically solving Kepler's equation
97  call solvekepler (m, ecc, e)
98 
99  ! Compute radial position and angular position (true anomaly) of reduced
100  ! mass
101  theta = 2.0*atan(sqrt((1+ecc)/(1-ecc))*tan(e/2))
102  rad = a*(1.0-ecc*ecc)/(1+ecc*cos(theta))
103 
104  ! Compute radial and angular velocities of reduced mass
105  thetadot = l/(mu*rad**2)
106  raddot = a*ecc*sin(theta)*(1-ecc**2)/(1+ecc*cos(theta))**2 * thetadot
107 
108  ! 2) Convert back to two-body problem
109 
110  ! Compute position of bodies
111  radx = rad*cos(theta)
112  rady = rad*sin(theta)
113  x1 = rx - m2/(m1+m2)*radx
114  y1 = ry - m2/(m1+m2)*rady
115  x2 = rx + m1/(m1+m2)*radx
116  y2 = ry + m1/(m1+m2)*rady
117 
118  ! Compute velocities of bodies
119  ! (in the center of momentum frame of reference)
120  vx = raddot*cos(theta)-rad*thetadot*sin(theta)
121  vy = raddot*sin(theta)+rad*thetadot*cos(theta)
122  vx1 = -m2/(m1+m2)*vx
123  vy1 = -m2/(m1+m2)*vy
124  vx2 = m1/(m1+m2)*vx
125  vy2 = m1/(m1+m2)*vy
126

128 ! vx1/KPS, vy1/KPS, sqrt(vx1**2+vy1**2)/KPS, &
129 ! x2/AU, y2/AU, sqrt(x2**2+y2**2)/AU, &
130 ! vx2/KPS, vy2/KPS, sqrt(vx2**2+vy2**2)/KPS
131 
132  end subroutine computebinary

134  ! ============================================
135  !> @brief Solves the Kepler equation, returning the eccentric anomaly
136  !> @details Solves the Kepler equation, viz.,
137  !! M = E - e*sin(E)
138  !! for E (eccentric anomaly), given M (mean anomaly). This uses a
139  !! Newton-Raphson solver with the specified relative tolerance
140  !> @params M Mean anomaly
141  !> @params E Eccentric anomaly
142 
143  subroutine solvekepler (M, ecc, E)
144 
145  implicit none
146 
147  real, intent(in) :: M
148  real, intent(in) :: ecc
149  real, intent(out) :: E
150 
151  real, parameter :: tol = 1e-10
152  real :: x, xn, rel_err
153 
154  if (m.eq.0.0) then
155  e = 0.0
156  return
157  end if
158 
159  if (ecc>0.8) then
160  x = sign(pi,m)
161  else
162  x = m
163  end if
164 
165  rel_err = tol*2.0
166  do while(rel_err.gt.tol)
167  xn = x - (x-ecc*sin(x)-m)/(1.0-ecc*cos(x))
168  if (x.ne.0.0) rel_err = abs((xn-x)/x)
169  x = xn
170  end do
171 
172  e = x
173  return
174 
175  end subroutine solvekepler
