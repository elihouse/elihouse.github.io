26 !> @brief Output simulation data in any format required
27 !> param noutput The output number
28 subroutine output (noutput)
29 
30  use parameters
31  use globals
32  use tictoc
33  implicit none
34 
35  integer, intent(in) :: noutput
36 
37  integer :: mark, startmark, p

44 
45  if (output_mode.eq.out_simult) then
46  write(logu,'(1x,a)') "Output mode: simultaneous"
47  else if (output_mode.eq.out_turns) then
48  write(logu,'(1x,a)') "Output mode: turn-based"
49  else
50  write(logu,'(1x,a,i0)') "Unrecognized data output mode: ", output_mode

52  call clean_abort(error_bad_output_mode)

55  if (.not.(output_bin.or.output_vtk)) then

58  write(logu,*) ""
59  return
60  end if
61 
62  call tic(startmark)
63 
64  do p=0,nprocs-1
65  if (rank.eq.p) then
66 
67  call tic(mark)

69  write(logu,*) ""
70 
71  ! Binary data format - writes flow vars (Us)

73  call writebin (noutput)
74  end if
75 
76  ! VTK format
77  if (output_vtk) then
78  call updateprims () ! Make sure primitives are updated

81  call write3dvtkgrid (noutput)
82  end if
83  end if
84

87  call writestate (noutput)
88  end if
89

91  " in ", nicetoc(mark)
92  write(logu,'(1x,a)') stamp()
93  write(logu,*) ""
94 
95  else
96 
97  if (output_mode.eq.out_turns) then
98  write(logu,'(1x,a,i0,a)') "Process ", p, " writing output ..."
99  write(logu,'(1x,a)') stamp()
100  write(logu,*) ""
101  call mpi_barrier (mpi_comm_world, ierr)
102  end if
103 
104  end if
105  end do
106 
107  call mpi_barrier (mpi_comm_world, ierr)
108  write(logu,'(1x,a,i0,a,a)') "> All ranks completed output ", noutput, &
109  " in ", nicetoc(startmark)
110  write(logu,'(1x,a)') stamp()
111  write(logu,*) ""
112
