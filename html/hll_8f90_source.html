26 !> @brief Harten, Lax, van Leer (HLL) Approximate Riemann Solver
27 !> @details Computes the intercell numerical fluxes for every cell interface
28 !! in a block using the HLL solver. This routine assumes the global vector of
29 !! primitives is up-to-date (including enough ghost cells) and exits with
30 !! computed values in the intercell numerical fluxes (FC, GC, HC). Interfaces
31 !! are defined to the right, that is:
32 !! FC(:,i,j,k): intercell flux between cell (i,j,k) and cell (i+1,j,k)
33 !! GC(:,i,j,k): intercell flux between cell (i,j,k) and cell (i,j+1,k)
34 !! HC(:,i,j,k): intercell flux between cell (i,j,k) and cell (i,j,k+1)
35 !! The order of the solver must be specified. First-order employs piece-wise
36 !! constant primitives in each cell, while second-ordr uses linear
37 !! reconstruction to interpolate the primitives at cell interfaces.
38 !> @param locIndx Local index of the block
39 !> @param order Order of solver (interpolation for primitives)
40 subroutine hllfluxes (locIndx, order)
41

44  implicit none
45

47  integer, intent(in) :: order

51  logical :: valid

53  ! First-order method

55 
56  ! -------------------------------------------
57 
58  case (1) ! First-order
59

61  do j=0,ncells_y

65  if (valid) then

90 
91  end if
92 
93  end do
94  end do
95  end do
96 
97  ! -------------------------------------------
98 
99  case (2) ! Second-order - requires limiter
100 
101  do i=0,ncells_x
102  do j=0,ncells_y
103  do k=0,ncells_z
104 
105  call validcell (i,j,k,valid)
106  if (valid) then
107

109  pll(:) = prim(locindx,:,i-1,j,k)
110  pl(:) = prim(locindx,:,i, j,k)

112  prr(:) = prim(locindx,:,i+2,j,k)
113  call limiter (pll,pl,pr,prr,limiter_type,neqtot)
114  call primfhll (pl, pr, ff)
115  fc(:,i,j,k) = ff(:)
116 
117  ! Y dimension
118  pll(:) = prim(locindx,:,i,j-1,k)
119  pl(:) = prim(locindx,:,i,j ,k)
120  pr(:) = prim(locindx,:,i,j+1,k)
121  prr(:) = prim(locindx,:,i,j+2,k)
122  call swapxy (pll)
123  call swapxy (pl)
124  call swapxy (pr)
125  call swapxy (prr)
126  call limiter (pll,pl,pr,prr,limiter_type,neqtot)
127  call primfhll (pl, pr, ff)
128  call swapxy (ff)
129  gc(:,i,j,k) = ff(:)
130 
131  ! Z dimension
132  pll(:) = prim(locindx,:,i,j,k-1)
133  pl(:) = prim(locindx,:,i,j,k )
134  pr(:) = prim(locindx,:,i,j,k+1)
135  prr(:) = prim(locindx,:,i,j,k+2)
136  call swapxz (pll)
137  call swapxz (pl)
138  call swapxz (pr)
139  call swapxz (prr)
140  call limiter (pll,pl,pr,prr,limiter_type,neqtot)
141  call primfhll (pl, pr, ff)
142  call swapxz (ff)
143  hc(:,i,j,k) = ff(:)
144 
145  end if

157 !> @brief HLL intercell fluxes along X using primitives at an interface
158 !> @param pL Vector of primitives at LEFT of interface
159 !> @param pR Vector of primitives at RIGHT of interface
160 !> @param ff Returned vector of HLL intercell fluxes
161 subroutine primfhll (pL, pR, ff)
162 
163  use parameters
164  implicit none
165 
166  real, intent(in) :: pL(neqtot)
167  real, intent(in) :: pR(neqtot)
168  real, intent(out) :: ff(neqtot)
169 
170  real :: sl, sr, sst
171  real :: uL(neqtot), uR(neqtot)
172  real :: fL(neqtot), fR(neqtot)
173 
174  ! Calculate wavespeeds
175  call wavespeed (pl, pr, sl, sr, sst)
176 
177 ! write(*,*) sl, sr
178 
179  ! Obtain intercell fluxes as given by 10.21 of Toro
180  if (sl.ge.0) then
181  call prim2fluxes (pl, dim_x, fl)
182  ff(:) = fl(:)
183  return
184  else if (sr.le.0) then
185  call prim2fluxes (pr, dim_x, fr)
186  ff(:) = fr(:)
187  return
188  else
189  call prim2fluxes (pl, dim_x, fl)
190  call prim2fluxes (pr, dim_x, fr)
191  call prim2flow (pl, ul)
192  call prim2flow (pr, ur)
193  ff(:) = (sr*fl(:)-sl*fr(:)+sl*sr*(ur(:)-ul(:)))/(sr-sl)
194  return
195  end if
196 
197 end subroutine primfhll
